\begin{boxe}
\textbf{Intermediate Value Theorem}: Suppose $f$ is continuous on a closed interval $[a,b]$. If $u$ is any number between $f(a)$ and $f(b)$, then there is at least one number $c$ in $[a,b]$ such that $f(c)=u$.
\end{boxe}
